    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         

       /*Declarations*/
       Map<Character,Double> Prob  = new HashMap<Character,Double>();
       double DoubleCodeWord = 0 , UpdatedCodeWord;
       String ans  = " ";
       int NumOfChars = 0;
       /* Probability input */
        File input = new File ("Proba.txt");
        Scanner filein = null;
        try {
            filein = new Scanner (input);
        } catch (FileNotFoundException ex) {
            Logger.getLogger(LZ77.class.getName()).log(Level.SEVERE, null, ex);
        }
         while (filein.hasNextLine()) { char c = filein.next().charAt(0); double cd = filein.nextDouble();  Prob.put(c, cd);}
         filein.close();
         
         /* CodeWord input */
        input = new File ("code.txt");
        filein = null;
        try {
            filein = new Scanner (input);
        } catch (FileNotFoundException ex) {
            Logger.getLogger(LZ77.class.getName()).log(Level.SEVERE, null, ex);
        }
         DoubleCodeWord = filein.nextDouble();
         NumOfChars =  filein.nextInt();
         filein.close();
         
        /* Building cummelative probabibilty */
        double Cum [][]  = new double [300][2];
        double prev=0;
        for (Map.Entry<Character,Double> entry : Prob.entrySet()) {
             char key = entry.getKey();
             double value = entry.getValue();
             Cum[(int)key][0]=prev;
              Cum[(int)key][1]=prev + value;
              prev =  Cum[(int)key][1];    
        }
        
        /* DeCompression Algo */
        double Lower = 0 , Upper=1 , range =1;
        UpdatedCodeWord = DoubleCodeWord;
        do {
            UpdatedCodeWord = (DoubleCodeWord - Lower) / (Upper - Lower);
            char FoundSymbol = 0;
            for (int i=65 ;i<256;i++) if (Cum[i][0] <= UpdatedCodeWord && Cum[i][1] >= UpdatedCodeWord ) { FoundSymbol = (char)i; break;}
            ans += FoundSymbol;
            range = Upper - Lower ;
            double OldLower = Lower;
            Lower = OldLower + range * Cum[(int)FoundSymbol][0];
            Upper = OldLower + range * Cum[(int)FoundSymbol][1];
            NumOfChars--;
        } while (NumOfChars !=0);
        
        
         /* Output to  Files */
       File outEncode = new File ("output.txt");
       PrintWriter o = null;
        try {
            o = new PrintWriter (outEncode);
        } catch (FileNotFoundException ex) {
            Logger.getLogger(LZ77.class.getName()).log(Level.SEVERE, null, ex);
        }
       o.println(ans);
       o.close();
       
        showMessageDialog(this,"Decompression is complete");
    }
