            while (filein.hasNextLine()){  tmp = filein.nextLine(); s+=tmp;}
            last = s.length();
            while (fence + take != last + 1)
            {
                sub = s.substring(fence, fence + take); // take from after fence
                searchString = s.substring(0, fence);
                pos = searchString.indexOf(sub);
                if (pos != -1)
                {
                    take++;
                    goback = fence - pos;
                }
                else
                {
                    len = take-1;
                    nxt = s.charAt(fence + len);
                    take = 1;
                    if (len == 0)
                    {
                        goback = 0;
                    }
                    output.println(goback + " " + len + " " + nxt );
                    fence += len + 1;
                }
                
            }
            char n = '0'; take--;
            output.println(goback + " " + take  + " " + n );
